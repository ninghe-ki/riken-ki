3 3. What is the average and median CO2 uptake of the plants from Quebec and Mississippi?

mean_co2 <- aggregate(uptake ~ Type, data = CO2, FUN = mean)
median_co2 <- aggregate(uptake ~ Type, data = CO2, FUN = median)

print(mean_co2)
         Type   uptake
1      Quebec 33.54286
2 Mississippi 20.88333
print(median_co2)
         Type uptake
1      Quebec  37.15
2 Mississippi  19.30
